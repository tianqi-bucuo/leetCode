给定一个字符串 (s) 和一个字符模式 (p) ，实现一个支持 '?' 和 '*' 的通配符匹配。

'?' 可以匹配任何单个字符。
'*' 可以匹配任意字符串（包括空字符串）。

```Java
public boolean isMatch(String s, String p) {
    int slen = s.length(), plen = p.length();
    //用于记录出现上一次出现'*'时，p和s匹配到的位置
    int sflag = -1, pflag = -1;
    int i = 0,j = 0;
    while(i < slen){
        if(j < plen && (p.charAt(j)=='?' || p.charAt(j)==s.charAt(i))){
            i++;j++;
        }else if(j < plen && p.charAt(j)=='*'){
            //作记录
            sflag = i;
            pflag = j;
            //从'*'匹配0个字符开始尝试，所以只加j
            j++;
        }else if (pflag == -1) {
            //如果匹配失败且之前没有出现过'*'，直接返回false
            return false;
        }else{
            //匹配失败但出现过'*'，尝试将'*'匹配多个字符
            i = ++sflag;
            j = pflag+1;
        }
    }
    //s串匹配完，这时只有p串剩下的都是'*'或也已匹配完才算匹配成功
    //因为'*'是从0个字符开始尝试匹配的
    for(;j < plen;j++)
        if(p.charAt(j)!='*') return false;
    return true;
}
```
