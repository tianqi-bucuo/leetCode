求把 N * M 的棋盘分割成若干个 1 * 2 的的长方形，有多少种方案。

***

```Java
public static long func(int m, int n) {
    long[][] f = new long[m + 1][1 << n];
    // 初始化(没有实际意义但很重要)
    f[0][0] = 1;

    // 检查状态是否合法：不能有连续的奇数个0
    boolean[] isValid = new boolean[1 << n];
    for (int i = 0; i < (1 << n); i++) {
        int cnt = 0, has = 0;
        isValid[i] = true;
        for (int j = 0; j < n; j++) {
            if ((i & (1 << j)) == 0) {
                cnt ^= 1;
            }else {
                has |= cnt;
                cnt = 0;
            }
        }
        isValid[i] = (has | cnt) == 0;
    }

    for (int i = 1; i <= m; i++) {
        for (int j = 0; j < (1 << n); j++) {
            f[i][j] = 0;
            for (int k = 0; k < (1 << n); k++) {
                if ((j & k) == 0 && isValid[j | k]) {
                    f[i][j] += f[i - 1][k];
                }
            }
        }
    }
    return f[m][0];
}
```
