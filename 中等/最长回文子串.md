给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。  

```Java
class Solution {
    public String longestPalindrome(String s) {
        char[] chs = s.toCharArray();
        int i = 0;
        int i1, j1;
        String res = "";
        for(;i < chs.length;i++){
            int j = chs.length-1;
            for(;j >= i;j--){
                if(chs[i] == chs[j]){
                    i1 = i+1;j1 = j-1;
                    while(j1>=i1 && chs[i1] == chs[j1]){
                        i1++;j1--;
                    }
                    if((i1 == j1+1 || i1 == j1+2) && s.substring(i, j+1).length() > res.length()){
                        res = s.substring(i, j+1);
                    }
                }
            }
        }
        return res;
    }
}
```
