给定一个数组 nums，编写一个函数将所有 0 移动到数组的末尾，同时保持非零元素的相对顺序。

```Java
public void moveZeroes(int[] nums) {
	if(nums == null || nums.length <= 1) {
		return;
	}
	// i指向非0，j指向0
	int j = 0;
	for(int i=0;i<nums.length;i++) {
		// 找到i后面第一个非0
		while(j < nums.length && (nums[j] == 0 || j < i)){
			j++;
		}
		// 说明i后面没有非0了，直接return
		if(j == nums.length)
			return;
		// 当前元素为0，和nums[j](非0)交换
		if(nums[i] == 0) {
			int tmp = nums[i];
			nums[i] = nums[j];
			nums[j++] = tmp;
		}
	}
}
```
