给出两个 非空 的链表用来表示两个非负的整数。其中，它们各自的位数是按照 逆序 的方式存储的，并且它们的每个节点只能存储 一位 数字。  

如果，我们将这两个数相加起来，则会返回一个新的链表来表示它们的和。  

您可以假设除了数字 0 之外，这两个数都不会以 0 开头。  

```Java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
                if (l1 == null){
            ListNode l3 = l2;
            return l3;
        }else if (l2 == null){
            ListNode l3 = l1;
            return l3;
        }else {
            int sum = l1.val+l2.val;
            ListNode l3 = new ListNode(sum%10);
            int carry = sum/10;
            ListNode curr = l3;
            l1 = l1.next;
            l2 = l2.next;
            while (l1 != null && l2 != null){
                sum = l1.val+l2.val+carry;
                curr.next = new ListNode(sum%10);
                curr = curr.next;
                carry = sum/10;
                l1 = l1.next;
                l2 = l2.next;
            }
            if (l1 == null && l2 != null){
                sum = carry+l2.val;
                carry = sum/10;
                curr.next = new ListNode(sum%10);
                curr = curr.next;
                l2 = l2.next;
                while (l2 != null){
                    sum = carry+l2.val;
                    carry = sum/10;
                    curr.next = new ListNode(sum%10);
                    curr = curr.next;
                    l2 = l2.next;
                }
                if(carry != 0){
                    curr.next = new ListNode(carry);
                }
            }else if (l1 != null){
                sum = carry+l1.val;
                carry = sum/10;
                curr.next = new ListNode(sum%10);
                curr = curr.next;
                l1 = l1.next;
                while (l1 != null){
                    sum = carry+l1.val;
                    carry = sum/10;
                    curr.next = new ListNode(sum%10);
                    curr = curr.next;
                    l1 = l1.next;
                }
                if (carry != 0){
                    curr.next = new ListNode(carry);
                }
            }else {
                if (carry != 0){
                    curr.next = new ListNode(carry);
                }
            }
            return l3;
        }
    }
}
```
